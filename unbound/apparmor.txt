#include <tunables/global>

profile unbound flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  file,
  signal,

  capability net_bind_service,
  capability setuid,
  capability setgid,
  capability sys_chroot,
  capability chown,

  # S6-Overlay & Bashio
  /init rix,
  /bin/** ix,
  /usr/bin/** ix,
  /etc/s6/** rix,
  /run/s6/** rwix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/** rwk,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rw,

  # Access to options and config
  /data/** rw,
  /addon_configs/unbound/** rw,

  # Transition to child profile for the daemon
  /usr/sbin/unbound cx,

  profile /usr/sbin/unbound flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>

    signal receive,

    capability net_bind_service,
    capability setgid,
    capability setuid,
    capability chown,
    capability sys_chroot,

    network inet dgram,
    network inet stream,
    network inet6 dgram,
    network inet6 stream,

    /usr/sbin/unbound rm,
    /usr/sbin/unbound-anchor rm,

    /etc/unbound/** rw,
    /addon_configs/unbound/** rw,
    /proc/** r,
    /dev/urandom r,
    /dev/null rw,
  }
}

#include <tunables/global>

profile unbound flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  file,
  signal (send) set=(kill,term,int,hup,cont),
  signal (receive) set=(kill,term,int,hup,cont,usr1,usr2),

  capability chown,

  # S6-Overlay & Bashio
  /init rix,
  /bin/** ix,
  /usr/bin/** ix,
  /etc/s6/** rix,
  /run/s6/** rwkix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/** rw,
  /usr/lib/bashio/** ix,
  /tmp/** rw,

  # Access to options and config
  /data/** rw,
  /addon_configs/unbound/** rwk,

  # Transition to child profile for the daemon
  /usr/sbin/unbound cx,

  profile /usr/sbin/unbound flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>

    signal (receive) set=(kill,term,int,hup,cont,usr1,usr2),

    capability net_bind_service,
    capability setgid,
    capability setuid,

    network inet dgram,
    network inet stream,
    network inet6 dgram,
    network inet6 stream,

    /usr/sbin/unbound rm,
    /etc/unbound/** r,
    /addon_configs/unbound/** rwk,
    /proc/** r,
    /dev/urandom r,
    /dev/null rw,
  }
}

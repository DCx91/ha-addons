#include <tunables/global>

profile aiostreams flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /bin/** ix,
  /command/** ix,
  /etc/s6-overlay/** rwix,
  /etc/services.d/** rwix,
  /init rix,
  /package/** rix,
  /run/s6/** rwkix,
  /usr/bin/** ix,

  # Bashio
  /tmp/** rwk,
  /usr/lib/bashio/** ix,

  # Addon data
  /data/** rw,

  # Node.js + pnpm
  /usr/local/bin/node ix,
  /usr/local/bin/pnpm ix,
  /usr/local/lib/** rm,
  /usr/local/share/** r,

  # AIOStreams application
  /app/** r,
  /app/node_modules/.pnpm/** rm,

  # OpenSSL
  /etc/ca-certificates/** r,
  /etc/ssl/** r,
  /usr/bin/openssl ix,
  /usr/share/ca-certificates/** r,

  # Proc access needed by Node.js
  @{PROC}/@{pid}/fd/ r,
  @{PROC}/@{pid}/stat r,

  # Network
  network inet stream,
  network inet6 stream,
}
